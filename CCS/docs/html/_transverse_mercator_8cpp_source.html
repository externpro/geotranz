199 void TransverseMercator::latLonToNorthingEasting(
200  const double &latitude,
201  const double &longitude,
202  double &northing,
203  double &easting )
204 {
205  // Convert longitude (Greenwhich) to longitude from the central meridian
206  // (-Pi, Pi] equivalent needed for checkLatLon.
207  // Compute its cosine and sine.
208  double lamda = longitude - TranMerc_Origin_Long;
209  if (lamda > PI)
210  lamda -= (2 * PI);
211  if (lamda < -PI)
212  lamda += (2 * PI);
213  checkLatLon( latitude, lamda );
214 
215  double cosLam = cos(lamda);
216  double sinLam = sin(lamda);
217  double cosPhi = cos(latitude);
218  double sinPhi = sin(latitude);
219 
220  double P, part1, part2, denom, cosChi, sinChi;
221  double U, V;
222  double c2ku[MAX_TERMS], s2ku[MAX_TERMS];
223  double c2kv[MAX_TERMS], s2kv[MAX_TERMS];
224 
225  // Ellipsoid to sphere
226  // --------- -- ------
227 
228  // Convert geodetic latitude, Phi, to conformal latitude, Chi
229  // Only the cosine and sine of Chi are actually needed.
230  P = exp(TranMerc_eps * aTanH(TranMerc_eps * sinPhi));
231  part1 = (1 + sinPhi) / P;
232  part2 = (1 - sinPhi) * P;
233  denom = part1 + part2;
234  cosChi = 2 * cosPhi / denom;
235  sinChi = (part1 - part2) / denom;
236 
237  // Sphere to first plane
238  // ------ -- ----- -----
239 
240  // Apply spherical theory of transverse Mercator to get (u,v) coord.s
241  U = aTanH(cosChi * sinLam);
242  V = atan2(sinChi, cosChi * cosLam);
243 
244  // Use trig identities to compute cosh(2kU), sinh(2kU), cos(2kV), sin(2kV)
245  computeHyperbolicSeries( 2.0 * U, c2ku, s2ku );
246  computeTrigSeries( 2.0 * V, c2kv, s2kv );
247 
248  // First plane to second plane
249  // Accumulate terms for X and Y
250  double xStar = 0;
251  double yStar = 0;
252 
253  for (int k = N_TERMS - 1; k >= 0; k--)
254  {
255  xStar += TranMerc_aCoeff[k] * s2ku[k] * c2kv[k];
256  yStar += TranMerc_aCoeff[k] * c2ku[k] * s2kv[k];
257  }
258 
259  xStar += U;
260  yStar += V;
261 
262  // Apply isoperimetric radius, scale adjustment, and offsets
263  easting = (TranMerc_K0R4 * xStar);
264  northing = (TranMerc_K0R4 * yStar);
265 }

329 void TransverseMercator::northingEastingToLatLon(
330  const double &northing,
331  const double &easting,
332  double &latitude,
333  double &longitude )
334 {
335  double c2kx[MAX_TERMS], s2kx[MAX_TERMS], c2ky[MAX_TERMS], s2ky[MAX_TERMS];
336  double U, V;
337  double lamda;
338  double sinChi;
339 
340  // Undo offsets, scale change, and factor R4
341  // ---- ------- ----- ------ --- ------ --
342  double xStar = TranMerc_K0R4inv * (easting);
343  double yStar = TranMerc_K0R4inv * (northing);
344 
345  // Use trig identities to compute cosh(2kU), sinh(2kU), cos(2kV), sin(2kV)
346  computeHyperbolicSeries( 2.0 * xStar, c2kx, s2kx );
347  computeTrigSeries( 2.0 * yStar, c2ky, s2ky );
348 
349  // Second plane (x*, y*) to first plane (u, v)
350  // ------ ----- -------- -- ----- ----- ------
351  U = 0;
352  V = 0;
353 
354  for (int k = N_TERMS - 1; k >= 0; k--)
355  {
356  U += TranMerc_bCoeff[k] * s2kx[k] * c2ky[k];
357  V += TranMerc_bCoeff[k] * c2kx[k] * s2ky[k];
358  }
359 
360  U += xStar;
361  V += yStar;
362 
363  // First plane to sphere
364  // ----- ----- -- ------
365  double coshU = cosh(U);
366  double sinhU = sinh(U);
367  double cosV = cos(V);
368  double sinV = sin(V);
369 
370  // Longitude from central meridian
371  if ((fabs(cosV) < 10E-12) && (fabs(coshU) < 10E-12))
372  lamda = 0;
373  else
374  lamda = atan2(sinhU, cosV);
375 
376  // Conformal latitude
377  sinChi = sinV / coshU;
378  latitude = geodeticLat( sinChi, TranMerc_eps );
379 
380  // Longitude from Greenwich
381  // -------- ---- ---------
382  longitude = TranMerc_Origin_Long + lamda;
383 }

395  ===----- ===---------
396  Algorithm developed by: C. Rollins April 18, 2006
397 
398  INPUT
399  -----
400  invfla Inverse flattening (reciprocal flattening)
401 
402  OUTPUT
403  ------
404  n1 Helmert's "n"

406  bBoeff Coefficients for chi as a trig series in omega

408 
409  EXPLANATIONS
410  ------------
411  omega is rectifying latitude
412  chi is conformal latitude
413  psi is geocentric latitude
414  phi is geodetic latitude, commonly, "the latitude"
415  R4 is the meridional isoperimetric radius
416  "a" is the semi-major axis of the ellipsoid
417  "b" is the semi-minor axis of the ellipsoid

419 
420  This calculation depends only on the shape of the ellipsoid and are
421  independent of the ellipsoid size.
422

652 double TransverseMercator::geodeticLat(
653  double sinChi,
654  double e )
655 {
656  double p;
657  double pSq;
658  double s_old = 1.0e99;
659  double s = sinChi;
660  double onePlusSinChi = 1.0+sinChi;
661  double oneMinusSinChi = 1.0-sinChi;
662 
663  for( int n = 0; n < 30; n++ )
664  {
665  p = exp( e * aTanH( e * s ) );
666  pSq = p * p;
667  s = ( onePlusSinChi * pSq - oneMinusSinChi )
668  /( onePlusSinChi * pSq + oneMinusSinChi );
669 
670  if( fabs( s - s_old ) < 1.0e-12 )
671  {
672  break;
673  }
674  s_old = s;
675  }
